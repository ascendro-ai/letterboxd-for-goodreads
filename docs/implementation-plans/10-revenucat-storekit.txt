import Foundation
import RevenueCat

@Observable
final class SubscriptionService: NSObject {
    static let shared = SubscriptionService()

    private(set) var isPremium = false
    private(set) var currentOffering: Offering?
    private(set) var activeSubscription: EntitlementInfo?
    private(set) var isLoading = false

    static let entitlementID = "premium"
    static let apiKey = "" // Set in RevenueCat dashboard → API Keys → public SDK key

    private let api = APIClient.shared

    private override init() {
        super.init()
    }

    // MARK: - Configuration

    func configure() {
        guard !Self.apiKey.isEmpty else { return }

        Purchases.logLevel = .debug
        Purchases.configure(
            with: .init(withAPIKey: Self.apiKey)
                .with(entitlementVerificationMode: .informational)
        )
        Purchases.shared.delegate = self
    }

    func configureUser(userID: String) {
        guard !Self.apiKey.isEmpty else { return }
        Purchases.shared.logIn(userID) { [weak self] customerInfo, _, error in
            if let error {
                AnalyticsService.captureError(error, context: ["action": "revenucat_login"])
                return
            }
            if let customerInfo {
                self?.updateStatus(from: customerInfo)
            }
        }
    }

    func logOut() {
        guard !Self.apiKey.isEmpty else { return }
        Purchases.shared.logOut { [weak self] customerInfo, error in
            if let error {
                AnalyticsService.captureError(error, context: ["action": "revenucat_logout"])
            }
            self?.isPremium = false
            self?.activeSubscription = nil
            self?.currentOffering = nil
        }
    }

    // MARK: - Check Status

    @MainActor
    func checkSubscriptionStatus() async {
        guard !Self.apiKey.isEmpty else { return }
        do {
            let customerInfo = try await Purchases.shared.customerInfo()
            updateStatus(from: customerInfo)
            await verifyWithBackend()
        } catch {
            AnalyticsService.captureError(error, context: ["action": "check_subscription"])
            isPremium = false
        }
    }

    // MARK: - Fetch Offerings

    @MainActor
    func fetchOfferings() async {
        guard !Self.apiKey.isEmpty else { return }
        isLoading = true
        defer { isLoading = false }

        do {
            let offerings = try await Purchases.shared.offerings()
            currentOffering = offerings.current
        } catch {
            AnalyticsService.captureError(error, context: ["action": "fetch_offerings"])
        }
    }

    // MARK: - Purchase

    @MainActor
    func purchase(package: Package) async throws -> Bool {
        let result = try await Purchases.shared.purchase(package: package)

        if result.userCancelled {
            return false
        }

        updateStatus(from: result.customerInfo)

        if isPremium {
            await verifyWithBackend()
        }

        return true
    }

    // MARK: - Restore Purchases

    @MainActor
    func restorePurchases() async throws {
        let customerInfo = try await Purchases.shared.restorePurchases()
        updateStatus(from: customerInfo)

        if isPremium {
            await verifyWithBackend()
        }
    }

    // MARK: - Premium Feature Checks

    var maxShelves: Int {
        isPremium ? .max : 20
    }

    var adsEnabled: Bool {
        !isPremium
    }

    var canCreateShelf: Bool {
        isPremium
    }

    func canCreateShelf(currentCount: Int) -> Bool {
        currentCount < maxShelves
    }

    // MARK: - Manage Subscription URL

    func openManageSubscriptions() async {
        guard let windowScene = await UIApplication.shared.connectedScenes
            .compactMap({ $0 as? UIWindowScene })
            .first else { return }

        do {
            try await AppStore.showManageSubscriptions(in: windowScene)
        } catch {
            if let url = URL(string: "https://apps.apple.com/account/subscriptions") {
                await UIApplication.shared.open(url)
            }
        }
    }

    // MARK: - Backend Verification

    private func verifyWithBackend() async {
        guard let appUserID = Purchases.shared.appUserID as String? else { return }

        do {
            let body = SubscriptionVerifyRequest(appUserID: appUserID)
            try await api.request(.post, path: "/me/subscription/verify", body: body)
        } catch {
            AnalyticsService.captureError(error, context: ["action": "verify_subscription_backend"])
        }
    }

    // MARK: - Private

    private func updateStatus(from customerInfo: CustomerInfo) {
        let entitlement = customerInfo.entitlements[Self.entitlementID]
        isPremium = entitlement?.isActive == true
        activeSubscription = entitlement
    }
}

// MARK: - PurchasesDelegate

extension SubscriptionService: PurchasesDelegate {
    nonisolated func purchases(_ purchases: Purchases, receivedUpdated customerInfo: CustomerInfo) {
        Task { @MainActor in
            updateStatus(from: customerInfo)
        }
    }
}

// MARK: - Request Model

struct SubscriptionVerifyRequest: Codable {
    let appUserID: String

    enum CodingKeys: String, CodingKey {
        case appUserID = "app_user_id"
    }
}

                        showSuccess = true

        if package.packageType == .annual {

        return nil
    }

struct PackageOptionView: View {
    let package: Package
    let isSelected: Bool
    let badge: String?
    let onTap: () -> Void

    init(package: Package, isSelected: Bool, badge: String? = nil, onTap: @escaping () -> Void) {
        self.package = package
        self.isSelected = isSelected
        self.badge = badge
        self.onTap = onTap
    }

    var body: some View {
        Button(action: onTap) {
            HStack {
                VStack(alignment: .leading, spacing: 2) {
                    HStack(spacing: 6) {
                        Text(package.storeProduct.localizedTitle)
                            .font(.subheadline.weight(.semibold))
                        if let badge {
                            Text(badge)
                                .font(.caption2.weight(.bold))
                                .padding(.horizontal, 6)
                                .padding(.vertical, 2)
                                .background(Color.accentColor)
                                .foregroundStyle(.white)
                                .clipShape(Capsule())
                        }
                    }
                    Text(packageDescription)
                        .font(.caption)
                        .foregroundStyle(.secondary)
                }
                Spacer()
                Image(systemName: isSelected ? "checkmark.circle.fill" : "circle")
                    .foregroundStyle(isSelected ? Color.accentColor : .secondary)
                    .font(.title3)
            }
            .padding()
            .background(isSelected ? Color.accentColor.opacity(0.1) : Color(.systemGray6))
            .clipShape(RoundedRectangle(cornerRadius: 12))
            .overlay(
                RoundedRectangle(cornerRadius: 12)
                    .stroke(isSelected ? Color.accentColor : .clear, lineWidth: 2)
            )
        }
        .buttonStyle(.plain)
    }

    private var packageDescription: String {
        let price = package.storeProduct.localizedPriceString
        switch package.packageType {
        case .monthly:
            return "\(price) per month"
        case .annual:
            let monthlyEquivalent = package.storeProduct.price / 12
            let formatter = NumberFormatter()
            formatter.numberStyle = .currency
            formatter.locale = package.storeProduct.priceFormatter?.locale ?? .current
            let monthlyString = formatter.string(from: monthlyEquivalent as NSDecimalNumber) ?? ""
            return "\(price) per year (\(monthlyString)/mo)"
        default:
            return price
        }
    }
}

from __future__ import annotations

import hashlib
import hmac
import logging
from uuid import UUID

import httpx
from sqlalchemy import update
from sqlalchemy.ext.asyncio import AsyncSession

from api.config import get_settings
from api.errors import AppError
from models import User

logger = logging.getLogger(__name__)

REVENUCAT_API_BASE = "https://api.revenuecat.com/v1"


async def verify_subscription(
    db: AsyncSession,
    user_id: UUID,
    app_user_id: str,
) -> bool:
    """Call RevenueCat API to verify the user's subscription status and update
    the is_premium flag in the database. Returns the new premium status."""
    settings = get_settings()

    if not settings.revenucat_api_key:
        raise AppError(500, "CONFIG_ERROR", "RevenueCat API key not configured")

    async with httpx.AsyncClient() as client:
        response = await client.get(
            f"{REVENUCAT_API_BASE}/subscribers/{app_user_id}",
            headers={
                "Authorization": f"Bearer {settings.revenucat_api_key}",
                "Content-Type": "application/json",
            },
            timeout=10.0,
        )

    if response.status_code == 404:
        logger.warning("RevenueCat subscriber not found: %s", app_user_id)
        await _update_premium_status(db, user_id, False)
        return False

    if response.status_code != 200:
        logger.error(
            "RevenueCat API error: status=%d body=%s",
            response.status_code,
            response.text,
        )
        raise AppError(502, "REVENUCAT_ERROR", "Failed to verify subscription with RevenueCat")

    data = response.json()
    subscriber = data.get("subscriber", {})
    entitlements = subscriber.get("entitlements", {})
    premium = entitlements.get("premium", {})

    is_premium = (
        premium.get("expires_date") is not None
        and premium.get("unsubscribe_detected_at") is None
    ) or premium.get("expires_date") is None and "premium" in entitlements

    # RevenueCat returns expires_date=null for lifetime, or a future date for active
    # Also check product_identifier exists to confirm actual entitlement
    if "premium" in entitlements:
        expires = premium.get("expires_date")
        if expires is None:
            # Lifetime entitlement
            is_premium = True
        else:
            # Active if expires_date is in the future (RevenueCat only returns
            # active entitlements, but we double-check)
            is_premium = True
    else:
        is_premium = False

    await _update_premium_status(db, user_id, is_premium)
    return is_premium


async def handle_webhook(
    db: AsyncSession,
    payload: dict,
    signature: str | None,
) -> None:
    """Process a RevenueCat server-to-server webhook event."""
    settings = get_settings()

    if settings.revenucat_webhook_secret and signature:
        if not _verify_webhook_signature(payload, signature, settings.revenucat_webhook_secret):
            raise AppError(401, "INVALID_SIGNATURE", "Webhook signature verification failed")

    event = payload.get("event", {})
    event_type = event.get("type", "")
    app_user_id = event.get("app_user_id", "")

    if not app_user_id:
        logger.warning("Webhook event missing app_user_id: %s", event_type)
        return

    logger.info("Processing RevenueCat webhook: type=%s user=%s", event_type, app_user_id)

    # Map RevenueCat event types to premium status changes
    grant_premium_events = {
        "INITIAL_PURCHASE",
        "RENEWAL",
        "UNCANCELLATION",
        "NON_RENEWING_PURCHASE",
    }
    revoke_premium_events = {
        "CANCELLATION",
        "EXPIRATION",
        "BILLING_ISSUE",
    }

    # app_user_id from RevenueCat matches user.id (set during configureUser)
    try:
        user_id = UUID(app_user_id)
    except ValueError:
        logger.warning("Invalid app_user_id (not a UUID): %s", app_user_id)
        return

    if event_type in grant_premium_events:
        await _update_premium_status(db, user_id, True)
        logger.info("Granted premium to user %s via %s", user_id, event_type)
    elif event_type in revoke_premium_events:
        # For CANCELLATION, the user retains access until period end.
        # EXPIRATION means the period actually ended.
        if event_type == "EXPIRATION" or event_type == "BILLING_ISSUE":
            await _update_premium_status(db, user_id, False)
            logger.info("Revoked premium from user %s via %s", user_id, event_type)
        elif event_type == "CANCELLATION":
            # Don't revoke yet - user still has access until expiration
            logger.info(
                "User %s cancelled but still active until period end", user_id
            )
    else:
        logger.debug("Ignoring RevenueCat event type: %s", event_type)


async def _update_premium_status(
    db: AsyncSession,
    user_id: UUID,
    is_premium: bool,
) -> None:
    """Update the user's is_premium flag in the database."""
    stmt = update(User).where(User.id == user_id).values(is_premium=is_premium)
    await db.execute(stmt)
    await db.commit()


def _verify_webhook_signature(
    payload: dict,
    signature: str,
    secret: str,
) -> bool:
    """Verify RevenueCat webhook signature using HMAC-SHA256."""
    import json

    body = json.dumps(payload, separators=(",", ":"), sort_keys=True).encode()
    expected = hmac.new(secret.encode(), body, hashlib.sha256).hexdigest()
    return hmac.compare_digest(expected, signature)

    app_user_id: str

    is_premium: bool


@router.post("/me/subscription/verify", response_model=VerifySubscriptionResponse)
async def verify_subscription(
    body: VerifySubscriptionRequest,
    db: DB,
    current_user: CurrentUser,
) -> VerifySubscriptionResponse:
    """Client-triggered subscription verification. Calls RevenueCat API to
    confirm the user's entitlement status and updates the database."""
    is_premium = await subscription_service.verify_subscription(
        db=db,
        user_id=current_user.id,
        app_user_id=body.app_user_id,
    )
    return VerifySubscriptionResponse(is_premium=is_premium)


class WebhookResponse(BaseModel):
    status: str = "ok"


@router.post("/webhooks/revenucat", response_model=WebhookResponse)
async def revenucat_webhook(
    request: Request,
    db: DB,
    x_revenucat_signature: Annotated[str | None, Header()] = None,
) -> WebhookResponse:
    """RevenueCat server-to-server webhook endpoint. This endpoint is public
    (no auth required) but validated via shared secret / signature.

    Configure this URL in RevenueCat Dashboard > Integrations > Webhooks:
    https://shelf-api.up.railway.app/api/v1/webhooks/revenucat
    """
    payload = await request.json()

    await subscription_service.handle_webhook(
        db=db,
        payload=payload,
        signature=x_revenucat_signature,
    )

    return WebhookResponse()
