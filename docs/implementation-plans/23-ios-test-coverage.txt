import Foundation
@testable import Shelf

/// Protocol-based mock that replaces APIClient for testing.
/// Each service method maps to a closure that tests can override.
final class MockAPIClient {
    /// Map of (method, path) → response data
    var responses: [String: Any] = [:]
    var requestLog: [(method: String, path: String, body: Any?)] = []

    /// Register a mock response for a given path
    func register<T: Codable>(path: String, response: T) {
        responses[path] = response
    }

    /// Register an error for a given path
    func registerError(path: String, error: Error) {
        responses[path] = error
    }

    func request<T: Codable>(
        method: HTTPMethod,
        path: String,
        body: (any Encodable)? = nil
    ) async throws -> T {
        requestLog.append((method: method.rawValue, path: path, body: body))

        guard let response = responses[path] else {
            throw APIError.notFound
        }

        if let error = response as? Error {
            throw error
        }

        guard let typed = response as? T else {
            throw APIError.decodingError
        }

        return typed
    }
}

/// API errors for test assertions
enum APIError: Error, Equatable {
    case notFound
    case unauthorized
    case serverError(Int)
    case decodingError
    case networkError
}

import XCTest
@testable import Shelf

final class BookTests: XCTestCase {

    func testDecodeBookFromJSON() throws {
        let data = MockData.bookJSON.data(using: .utf8)!
        let decoder = JSONDecoder()
        decoder.keyDecodingStrategy = .convertFromSnakeCase
        let book = try decoder.decode(Book.self, from: data)

        XCTAssertEqual(book.id, MockData.workID)
        XCTAssertEqual(book.title, "The Great Gatsby")
        XCTAssertEqual(book.authors, ["F. Scott Fitzgerald"])
        XCTAssertEqual(book.firstPublishedYear, 1925)
        XCTAssertEqual(book.averageRating, 4.2, accuracy: 0.01)
        XCTAssertEqual(book.ratingsCount, 1500)
        XCTAssertNil(book.coverImageURL)
    }

    func testDecodeBookWithCoverURL() throws {
        let json = """
        {
            "id": "33333333-3333-3333-3333-333333333333",
            "title": "Test Book",
            "authors": ["Author"],
            "description": null,
            "first_published_year": null,
            "cover_image_url": "https://covers.shelf.app/thumb/abc.webp",
            "average_rating": null,
            "ratings_count": 0,
            "subjects": []
        }
        """
        let data = json.data(using: .utf8)!
        let decoder = JSONDecoder()
        decoder.keyDecodingStrategy = .convertFromSnakeCase
        let book = try decoder.decode(Book.self, from: data)

        XCTAssertEqual(book.coverImageURL, "https://covers.shelf.app/thumb/abc.webp")
        XCTAssertNil(book.description)
        XCTAssertNil(book.firstPublishedYear)
        XCTAssertNil(book.averageRating)
    }

    func testEncodeBookRoundTrip() throws {
        let book = MockData.makeBook()
        let encoder = JSONEncoder()
        encoder.keyEncodingStrategy = .convertToSnakeCase
        let data = try encoder.encode(book)

        let decoder = JSONDecoder()
        decoder.keyDecodingStrategy = .convertFromSnakeCase
        let decoded = try decoder.decode(Book.self, from: data)

        XCTAssertEqual(decoded.id, book.id)
        XCTAssertEqual(decoded.title, book.title)
    }
}

================================================================================
MAKING SERVICES TESTABLE (Prerequisite Refactoring)
================================================================================

The current architecture uses singleton .shared instances (e.g.,
BookService.shared, APIClient.shared) which makes unit testing difficult.

Recommended refactoring for testability:

1. Extract protocols for each service:

    protocol BookServiceProtocol {
        func search(query: String, cursor: String?) async throws -> PaginatedResponse<Book>
        func getDetail(id: UUID) async throws -> BookDetail
    }

    final class BookService: BookServiceProtocol { ... }

2. Accept service dependencies in view model initializers:

    @Observable
    final class SearchViewModel {
        private let bookService: BookServiceProtocol

        init(bookService: BookServiceProtocol = BookService.shared) {
            self.bookService = bookService
        }
    }

3. In tests, inject mock implementations:

    final class MockBookService: BookServiceProtocol {
        var searchResult: PaginatedResponse<Book> = ...
        func search(...) async throws -> PaginatedResponse<Book> {
            return searchResult
        }
    }

This refactoring is optional but significantly improves test coverage
quality. Without it, tests are limited to model decoding and integration
tests that hit real/mocked HTTP endpoints.
