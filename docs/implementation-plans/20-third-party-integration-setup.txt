1. SUPABASE AUTH (Apple, Google, Email Sign In)
================================================================================

Status: Backend JWT validation works. iOS AuthService scaffolded. Need real
Supabase project + OAuth provider configuration.

────────────────────────────────────────────────────────────────────────────────
SETUP STEPS:
────────────────────────────────────────────────────────────────────────────────

a) Create Supabase project at https://supabase.com/dashboard
   - Note: Project URL, anon key, service role key, JWT secret

b) Enable auth providers in Supabase dashboard → Authentication → Providers:

   APPLE SIGN IN:
   - Register App ID at developer.apple.com → Certificates, Identifiers & Profiles
   - Enable "Sign In with Apple" capability
   - Create a Services ID for web (Supabase redirect URL)
   - Generate a private key (.p8) for Sign In with Apple
   - In Supabase: paste Client ID (Services ID), paste key, set team ID + key ID
   - In Xcode: add "Sign in with Apple" capability to Shelf.entitlements

   GOOGLE SIGN IN:
   - Create project at console.cloud.google.com
   - Enable Google Identity API
   - Create OAuth 2.0 Client ID (iOS type) — needs bundle ID
   - Create OAuth 2.0 Client ID (Web type) — for Supabase
   - In Supabase: paste Web Client ID + Client Secret
   - In Xcode: add reversed client ID to URL Schemes in Info.plist

   EMAIL + PASSWORD:
   - Enabled by default in Supabase
   - Configure email templates in Supabase → Authentication → Email Templates
   - Set up custom SMTP (optional) for branded emails

c) Set environment variables:

   Backend (.env):
     SUPABASE_URL=https://xxxx.supabase.co
     SUPABASE_SERVICE_KEY=eyJ...  (service role key, NOT anon key)
     SUPABASE_JWT_SECRET=your-jwt-secret  (Settings → API → JWT Secret)

   iOS (APIClient.swift or environment config):
     Update AuthService.swift with:
       - supabaseURL = "https://xxxx.supabase.co"
       - supabaseAnonKey = "eyJ..." (anon key, safe for client)

"""Webhook receivers for third-party services."""

from __future__ import annotations

import hmac
import hashlib
import logging

from fastapi import APIRouter, Header, HTTPException, Request

from backend.api.deps import DB
from backend.api.config import get_settings

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/webhooks", tags=["webhooks"])


@router.post("/revenuecat")
async def revenuecat_webhook(
    request: Request,
    db: DB,
    authorization: str | None = Header(None),
) -> dict[str, str]:
    """Handle RevenueCat subscription events.

    Events: INITIAL_PURCHASE, RENEWAL, CANCELLATION, EXPIRATION, etc.
    Updates the user's is_premium flag in our database.
    """
    settings = get_settings()

    # Verify webhook auth header
    if settings.revenuecat_webhook_secret:
        if authorization != f"Bearer {settings.revenuecat_webhook_secret}":
            raise HTTPException(status_code=401, detail="Invalid webhook auth")

    body = await request.json()
    event_type = body.get("event", {}).get("type", "")
    app_user_id = body.get("event", {}).get("app_user_id", "")

    if not app_user_id:
        return {"status": "ignored"}

    # Update premium status based on event type
    from sqlalchemy import update
    from backend.api.model_stubs import User

    grant_premium = event_type in (
        "INITIAL_PURCHASE", "RENEWAL", "PRODUCT_CHANGE",
        "UNCANCELLATION", "SUBSCRIBER_ALIAS",
    )
    revoke_premium = event_type in (
        "EXPIRATION", "BILLING_ISSUE",
    )

    if grant_premium or revoke_premium:
        await db.execute(
            update(User)
            .where(User.id == app_user_id)
            .values(is_premium=grant_premium)
        )
        logger.info("RevenueCat %s for user %s → premium=%s",
                     event_type, app_user_id, grant_premium)

    return {"status": "ok"}


Register in main.py:

    from backend.api.routes import webhooks
    app.include_router(webhooks.router, prefix="/api/v1")

import GoogleSignIn

func signInWithGoogle() async throws {
    guard let windowScene = UIApplication.shared.connectedScenes.first as? UIWindowScene,
          let rootVC = windowScene.windows.first?.rootViewController else {
        throw AuthError.noRootViewController
    }

    let result = try await GIDSignIn.sharedInstance.signIn(withPresenting: rootVC)
    guard let idToken = result.user.idToken?.tokenString else {
        throw AuthError.missingToken
    }

    // Pass Google ID token to Supabase for session creation
    try await signInWithProvider(
        provider: "google",
        idToken: idToken,
        accessToken: result.user.accessToken.tokenString
    )
}
