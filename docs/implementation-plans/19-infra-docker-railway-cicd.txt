# Multi-stage build for small production image
FROM python:3.12-slim AS base

ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1

WORKDIR /app

# Install system deps for asyncpg (libpq) and cryptography
RUN apt-get update && \
    apt-get install -y --no-install-recommends libpq-dev gcc && \
    rm -rf /var/lib/apt/lists/*

COPY backend/requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY backend/ .

# Run Alembic migrations on startup, then launch uvicorn
# The entrypoint script handles migration + startup sequencing
COPY infra/docker/entrypoint-backend.sh /entrypoint.sh
RUN chmod +x /entrypoint.sh

EXPOSE 8000

ENTRYPOINT ["/entrypoint.sh"]
CMD ["uvicorn", "api.main:app", "--host", "0.0.0.0", "--port", "8000", "--workers", "2"]

# Railway Cron Job Configuration

Configure these in Railway dashboard → Service → Settings → Cron:

| Schedule        | Command                                      | Description              |
|-----------------|----------------------------------------------|--------------------------|
| 0 3 * * *       | python -m sync.nightly_sync                  | OL recent changes sync   |
| 0 4 * * *       | python -m sync.taste_match_job               | Recompute taste scores   |
| 0 5 * * *       | python -m cover_processing.fetch_covers      | Backfill missing covers  |

All times are UTC. Nightly sync must run before taste match (taste match
uses updated rating data from the sync).

"""Health check endpoint for Railway deployment monitoring."""

from __future__ import annotations

from fastapi import APIRouter
from sqlalchemy import text

from backend.api.deps import DB

router = APIRouter(tags=["health"])


@router.get("/health")
async def health_check(db: DB) -> dict[str, str]:
    """Basic health check — verifies DB connectivity."""
    await db.execute(text("SELECT 1"))
    return {"status": "healthy"}


Register in backend/api/main.py alongside other routers:

    from backend.api.routes import health
    app.include_router(health.router)

FILE: .github/workflows/backend-ci.yml
────────────────────────────────────────────────────────────────────────────────

name: Backend CI

on:
  push:
    branches: [main]
    paths: ["backend/**"]
  pull_request:
    branches: [main]
    paths: ["backend/**"]

jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      - run: pip install ruff
      - run: ruff check backend/
      - run: ruff format --check backend/

  test:
    runs-on: ubuntu-latest
    needs: lint
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
          cache: pip
          cache-dependency-path: backend/requirements.txt
      - run: pip install -r backend/requirements.txt
      - name: Run tests
        working-directory: backend
        run: pytest tests/ -v --tb=short --junitxml=test-results.xml
        env:
          DATABASE_URL: "sqlite+aiosqlite://"
          SUPABASE_JWT_SECRET: "test-secret-key-minimum-32-characters!!"
          ENVIRONMENT: test
      - uses: actions/upload-artifact@v4
        if: always()
        with:
          name: backend-test-results
          path: backend/test-results.xml

name: Deploy to Railway

on:
  push:
    branches: [main]

jobs:
  deploy-backend:
    runs-on: ubuntu-latest
    if: contains(github.event.head_commit.modified, 'backend/')
    steps:
      - uses: actions/checkout@v4
      - name: Deploy Backend to Railway
        uses: bervProject/railway-deploy@main
        with:
          railway_token: ${{ secrets.RAILWAY_TOKEN }}
          service: shelf-backend

  deploy-pipeline:
    runs-on: ubuntu-latest
    if: contains(github.event.head_commit.modified, 'pipeline/')
    steps:
      - uses: actions/checkout@v4
      - name: Deploy Pipeline to Railway
        uses: bervProject/railway-deploy@main
        with:
          railway_token: ${{ secrets.RAILWAY_TOKEN }}

name: iOS CI

on:
  push:
    branches: [main]
    paths: ["ios/**"]
  pull_request:
    branches: [main]
    paths: ["ios/**"]

jobs:
  build-and-test:
    runs-on: macos-14
    steps:
      - uses: actions/checkout@v4
      - name: Select Xcode
        run: sudo xcode-select -s /Applications/Xcode_15.4.app
      - name: Build
        working-directory: ios/Shelf
        run: |
          xcodebuild build-for-testing \
            -project Shelf.xcodeproj \
            -scheme Shelf \
            -destination "platform=iOS Simulator,name=iPhone 15,OS=17.5" \
            -quiet
      - name: Test
        working-directory: ios/Shelf
        run: |
          xcodebuild test \
            -project Shelf.xcodeproj \
            -scheme Shelf \
            -destination "platform=iOS Simulator,name=iPhone 15,OS=17.5" \
            -quiet \
            -resultBundlePath TestResults.xcresult
      - uses: actions/upload-artifact@v4
        if: always()
        with:
          name: ios-test-results
          path: ios/Shelf/TestResults.xcresult

================================================================================
DEPLOYMENT CHECKLIST (first-time setup)
================================================================================

1. Create Railway project with two services: shelf-backend, shelf-pipeline
2. Attach Railway Postgres plugin to both services
3. Set all environment variables (see above)
4. Upload APNs .p8 key file to backend service (Railway volume or build secret)
5. Create GitHub repo secret: RAILWAY_TOKEN
6. Push to main — CI runs tests, deploy workflow triggers Railway
7. Verify health check: curl https://shelf-backend.up.railway.app/health
8. Run initial migration: Railway CLI → railway run alembic upgrade head
9. Run bulk import: Railway CLI → railway run python -m import_ol.bulk_import ...
10. Enable cron jobs in Railway dashboard
